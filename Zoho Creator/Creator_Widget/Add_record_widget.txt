  <script>
    document.addEventListener("DOMContentLoaded", function () {
      ZOHO.CREATOR.init().then(function () {
        console.log("Zoho Creator Widget SDK Initialized");

        document.getElementById("agentForm").addEventListener("submit", async function (event) {
          event.preventDefault();

          const phoneInput = document.getElementById("phone").value.trim();
          const phonePattern = /^\d{10}$/;

          if (!phonePattern.test(phoneInput)) {
            alert("Please enter a valid 10-digit phone number.");
            return;
          }

          const config = {
            appName: "real-estate-management",
            formName: "Agent_Details",
            data: {
              "data": {
                "Agent_Name": document.getElementById("name").value,
                "Email": document.getElementById("email").value,
                "Experience_Years": document.getElementById("experience").value,
                "Phone_Number1": phoneInput
              }
            }
          };

          ZOHO.CREATOR.API.addRecord(config).then(function (response) {
            if (response.code === 3000) {
              console.log('Record added successfully:', response);
              alert('Agent created successfully!');
              document.getElementById("agentForm").reset(); // Clear form
            } else {
              console.error('Error adding record:', response);
              alert('Failed to create agent.');
            }
          });
        });
      });
    });
  </script>
